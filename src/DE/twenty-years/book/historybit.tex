% This defines a mdframed environment called ``historybit''.  This
% environment is a colored frame that can span multiple pages.
%
% The title of the frame is shown as ``History bit N: yourtitle''.
%
% You can use it like
% \begin{historybit}{When we were young}
% When we were young we didn't know nothing!
% \end{historybit}

\definecolor{HistoryBitBackground}{gray}{0.85}

\mdfdefinestyle{theoremstyle}{
  hidealllines=true,                         % no frame
  backgroundcolor=HistoryBitBackground,
  skipabove=\baselineskip,                   % space above box
  skipbelow=\baselineskip,                   % space below box
  innerleftmargin=10pt,
  innerrightmargin=10pt,
  innerbottommargin=10pt,
  innertopmargin=\baselineskip,
  frametitleaboveskip=10pt,
  frametitlebelowskip=0,
}

\mdtheorem[style=theoremstyle]{historybit}{History bit}
